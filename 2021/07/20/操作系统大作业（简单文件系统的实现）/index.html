简单文件系统的实现

这玩意儿是操作系统的课设，大概意思就是根据FAT的思路来完成一个简单的文件系统。

内容要求

  1. 在内存中开辟一个虚拟磁盘空间作为文件存储分区，在其上实现一个简单的基于多级目录的单用户单任务系统中的文件系统。在退出该文件系统的使用时，应将该虚拟文件系统以一个Windows 文件的方式保存到磁盘上，以便下次可以再将它恢复到内存的虚拟磁盘空间中。
  2. 文件存储空间的分配可采用显式链接分配或其他的办法
  3. 空闲磁盘空间的管理可选择位示图或其他的办法。如果采用位示图来管理文件存储空间，并采用显式链接分配方式，那么可以将位示图合并到FAT中
  4. 文件目录结构采用多级目录结构。为了简单起见，可以不使用索引结点，其中的每个目录项应包含文件名、物理地址、长度等信息，还可以通过目录项实现对文件的读和写的保护。

#include<bits/stdc++.h>
using namespace std;
       
#define  BLOCK_SIZE         1024        //磁盘块大小
#define  SIZE               1024000     //虚拟磁盘空间大小
#define  END                65535       //FAT中的文件结束标志
#define  FREE               0           //FAT中盘块空闲标志
#define  ROOT_BLOCK_NUM     2           //根目录区所占盘块总数
#define  MAX_OPEN_FILE      10          //最多同时打开文件个数

typedef struct TIME{
    int hour;
    int minute;
    int second;
} my_time;


typedef struct DATE{
    int year;
    int month;
    int day;
} my_date;


typedef struct FCB {
    char filename[8];                   //文件名
    char extname[3];                    //文件仿照 FAT16 设置的
    unsigned char attribute;            //文件属性字段 值为 0 时表示目录文件，值为 1 时表示数据文件
    my_time time;                       //文件创建时间
    my_date date;                       //文件创建日期
    unsigned short first;               //起始盘块号
    unsigned long length;               //文件长度
    char free;                          //表示目录项是否为空，若值为 0，表示空，值为 1，表示已分配
} fcb;


typedef struct FAT {
    unsigned short id;
} fat;


typedef struct USER_OPEN {
    fcb open_file_fcb;
    int dir_no;                         //相应打开文件的目录项在父目录文件中的盘块号
    int dir_off;                        //相应打开文件的目录项在父目录文件的中的偏移地址
    char dir[MAX_OPEN_FILE][80];        //相应打开文件所在的目录名
    int count;                          //读写指针在文件中的位置
    char fcb_state;                     //是否修改了文件的 FCB 的内容
    char t_openfile;                    //表示该用户打开表项是否为空，若值为0，表示为空，否则表示已被某打开文件占据
} user_open;


typedef struct BLOCK0 {
    char magic_number[8];               //文件系统的魔数
    char information[200];              //存储一些描述信息，如磁盘块大小、磁盘块数量、最多打开文件数等
    unsigned short root;                //根目录文件的起始盘块号
    unsigned char *start_block;         //虚拟磁盘上数据区开始位置
} block0;

int do_read(int fd, int len, char *text) {
    unsigned char *buffer = (unsigned char *) malloc(BLOCK_SIZE);
    if (buffer == NULL) {
        puts("Error(Do_Read): buffer can't be get!");
        return -1;
    }

    int required_len = len;
    char *text_ptr = text;
    int block_num = open_file_list[fd].open_file_fcb.first;
    int offset = open_file_list[fd].count;
    fat *fat_ptr = (fat *) (my_v_hard + BLOCK_SIZE) + block_num;

    while (offset >= BLOCK_SIZE) {
        offset -= BLOCK_SIZE;
        block_num = fat_ptr->id;
        if (block_num == END) {
            puts("Error(Do_Read): target don't exist");
            return -1;
        }
        fat_ptr = (fat *) (my_v_hard + BLOCK_SIZE) + block_num;
    }

    unsigned char *block_ptr = my_v_hard + block_num * BLOCK_SIZE;
    memcpy(buffer, block_ptr, BLOCK_SIZE);

    while (len > 0) {
        if (BLOCK_SIZE - offset > len) {
            memcpy(text_ptr, buffer + offset, len);
            text_ptr += len;
            open_file_list[fd].count += len;
            len = 0;
            break;
        } else {
            memcpy(text_ptr, buffer + offset, BLOCK_SIZE - offset);
            text_ptr += BLOCK_SIZE - offset;
            open_file_list[fd].count  += (BLOCK_SIZE  - offset);
            len -= BLOCK_SIZE - offset;
        }
        //重新计算指针位置
        offset = open_file_list[fd].count;
        block_num = open_file_list[fd].open_file_fcb.first;
        fat_ptr = (fat *) (my_v_hard + BLOCK_SIZE) + block_num;
        block_ptr = my_v_hard + block_num * BLOCK_SIZE;
        while (offset >= BLOCK_SIZE) {
            offset -= BLOCK_SIZE;
            block_num = fat_ptr->id;
            if (block_num == END) {
                puts("Error(Do_Read): target don't exist");
                return -1;
            }
            fat_ptr = (fat *) (my_v_hard + BLOCK_SIZE) + block_num;
        }
        block_ptr = my_v_hard + block_num * BLOCK_SIZE;
        memcpy(buffer, block_ptr, BLOCK_SIZE);
    }

    free(buffer);
    return required_len - len;
}
void my_format() {
    boot = (block0*)my_v_hard;
    time_t now;
    struct tm now_time;
    time(&now);
    localtime_s(&now_time, &now);
    strcpy_s(boot->magic_number, sizeof("10101010"), "10101010");
    strcpy_s(boot->information, sizeof("nothing"), "nothing");
    boot->root = 5;
    boot->start_block = my_v_hard + BLOCK_SIZE * 5;    //一个引导盘，四个fat表盘

    fat* fat1 = (fat*) (my_v_hard + BLOCK_SIZE);
    fat* fat2 = (fat*) (my_v_hard + 3 * BLOCK_SIZE);

    for(int i = 0; i < 5; i ++) {       //处理fat表
        fat2[i].id = END;
        fat1[i].id = END;
    }
    fat1[5].id = 6;
    fat2[5].id = 6;
    fat1[6].id = END;
    fat2[6].id = END;

    for (int i = 7; i < (SIZE / BLOCK_SIZE); i ++) {    
        fat2[i].id = FREE;
        fat1[i].id = FREE;
    }

    //处理根目录
    fcb* ptr_current_dir_fcb = (fcb*)(my_v_hard + BLOCK_SIZE * 5);
    strcpy_s(ptr_current_dir_fcb->filename, sizeof("."), ".");
    strcpy_s(ptr_current_dir_fcb->extname, sizeof(""), "");
    ptr_current_dir_fcb->attribute = 0;
    ptr_current_dir_fcb->time.hour = now_time.tm_hour;
    ptr_current_dir_fcb->time.minute = now_time.tm_min;
    ptr_current_dir_fcb->time.second = now_time.tm_sec;
    ptr_current_dir_fcb->date.year = now_time.tm_year;
    ptr_current_dir_fcb->date.month = now_time.tm_mon;
    ptr_current_dir_fcb->date.day = now_time.tm_mday;
    ptr_current_dir_fcb->free = 1;
    ptr_current_dir_fcb->first = 5;
    ptr_current_dir_fcb->length = 2 * sizeof(fcb);
    ptr_current_dir_fcb->free = 1;

    fcb *ptr_next_dir_fcb = ptr_current_dir_fcb + 1;
    memcpy(ptr_next_dir_fcb, ptr_current_dir_fcb, sizeof(fcb));

    for (int i = 2; i < (int) (BLOCK_SIZE / sizeof(fcb)); i ++) {
        ptr_next_dir_fcb ++;
        strcpy_s(ptr_next_dir_fcb->filename, sizeof(""), "");
        ptr_next_dir_fcb->free = 0;
    }
    return ;
}

void init_sys() {
    if (fopen_s(&fptr, SYS_FILE, "w") != 0) {
        printf("init error! can't open file\n");
        return;
    }
    printf("file sys is not exist! now create it!\n");
    my_format();
    return;
}

void start_sys() {
    unsigned char buffer[SIZE];
    unsigned char st[8] = {'1', '0', '1', '0', '1', '0', '1', '0'};
    my_v_hard = (unsigned char*)malloc(SIZE);

    if (fopen_s(&fptr, SYS_FILE, "r") == 0) {   //比较系统魔数并初始化
        fread(buffer, SIZE, 1, fptr);
        if (memcmp(st, buffer, sizeof(st)) == 0) {
            memcpy(my_v_hard, buffer, SIZE);
        }
        else init_sys();
    }
    else init_sys();
    fwrite(my_v_hard, SIZE, 1, fptr);
    fclose(fptr);

    fcb *root = (fcb*)(my_v_hard + BLOCK_SIZE * 5);
    user_open *current = &open_file_list[0];
    strcpy_s(current->open_file_fcb.filename, sizeof(root->filename), root->filename);
    strcpy_s(current->open_file_fcb.extname, sizeof(root->extname), root->extname);
    current->open_file_fcb.attribute = root->attribute;
    current->open_file_fcb.time = root->time;
    current->open_file_fcb.date = root->date;
    current->open_file_fcb.first = root->first;
    current->open_file_fcb.length = root->length;
    current->open_file_fcb.free = root->free;
    current->dir_no = 5;
    current->dir_off = 0;
    strcpy_s(current->dir[0], sizeof("/"), "/");
    current->count = 0;
    current->fcb_state = 0;
    current->t_openfile = 1;
    return ;
}

void exitsys() {
    //关闭所有打开的文件
    //将虚拟磁盘的内容保存到文件中
    while(ptr_current_dir - open_file_list) {
        my_close(ptr_current_dir - open_file_list);
    }
    if (fopen_s(&fptr, SYS_FILE, "W") != 0) {   //无法打开文件
        printf("error in exitsys!!\n");
        return;
    }   
    fwrite(my_v_hard, SIZE, 1, fptr);
    fclose(fptr);
    printf("file sys is closed!!\n");
    return;
}

void my_close(int fd) {
    if (fd < 0 || fd >= MAX_OPEN_FILE) {
        printf("error in my_close!the file don't exist\n");
        return;
    }
    // 关闭文件的时候需要检查fcb是否被修改
    // 需要改变的是父目录的磁盘内容
    int father_fd = - 1;
    for(int i = 0; i < MAX_OPEN_FILE; i ++) {
        if (open_file_list[i].open_file_fcb.first == open_file_list[fd].dir_no) {
            father_fd = i;
            break;
        }
    }
    if (fd == - 1) {
        printf("error in my_close: father file is not exist!\n");
        return;
    }

    static char buffer[SIZE];
    fcb* fcb_ptr;   

    if (open_file_list[fd].fcb_state == 1) {    
        do_read(father_fd, (int)open_file_list[father_fd].open_file_fcb.length, buffer);
        int fa_off = open_file_list[fd].dir_off;
        fcb_ptr = (fcb*) (buffer + sizeof(fcb) * fa_off);

        strcpy_s(fcb_ptr->extname, sizeof(open_file_list[fd].open_file_fcb.extname), open_file_list[fd].open_file_fcb.extname);
        strcpy_s(fcb_ptr->filename, sizeof(open_file_list[fd].open_file_fcb.filename), open_file_list[fd].open_file_fcb.filename);
        fcb_ptr->first = open_file_list[fd].open_file_fcb.first;
        fcb_ptr->free = open_file_list[fd].open_file_fcb.free;
        fcb_ptr->length = open_file_list[fd].open_file_fcb.length;
        fcb_ptr->time = open_file_list[fd].open_file_fcb.time;
        fcb_ptr->date = open_file_list[fd].open_file_fcb.date;
        fcb_ptr->attribute = open_file_list[fd].open_file_fcb.attribute;
        open_file_list[father_fd].count = fa_off * int(sizeof(fcb));

        do_write(father_fd, (char*)fcb_ptr, sizeof(fcb), 2);

    }
    memset(&open_file_list[fd], 0, sizeof(USER_OPEN));
    ptr_current_dir = open_file_list + father_fd;   //当前文件指针打到父目录
    return;
}

int do_write(int fd, char* text, int len, char wstyle) {
    unsigned char* buffer_now = (unsigned char*)malloc(BLOCK_SIZE);
    if (buffer_now == NULL) {
        printf("error in do_write!");
        return -1;
    }

    int block_pos = open_file_list[fd].open_file_fcb.first;
    char* text_ptr = text;
    fat *fat_ptr = (fat*) (my_v_hard + BLOCK_SIZE) + block_pos;

    // 1.截断写 2.覆盖写 3.追加写
    // prework
    if (wstyle == 1) {
        open_file_list[fd].count = 0;
        open_file_list[fd].open_file_fcb.length = 0;
    }
    else if (wstyle == 3) {
        open_file_list[fd].count = (int)(open_file_list[fd].open_file_fcb.length);
        if (open_file_list[fd].open_file_fcb.attribute == 1) {
            if (open_file_list[fd].open_file_fcb.length != 0) {
                open_file_list[fd].count = (int)open_file_list[fd].open_file_fcb.length - 1;
            }
        }
    }

    int offset = open_file_list[fd].count;
    while(offset >= BLOCK_SIZE) {
        block_pos = fat_ptr->id;
        if (block_pos == END) {
            printf("error in do_write.\n");
            return -1;
        }
        fat_ptr = (fat*) (my_v_hard + BLOCK_SIZE) + block_pos;
        offset -= BLOCK_SIZE;
    }

    int already_count = 0;
    unsigned char* block_ptr = (unsigned char*)(my_v_hard + BLOCK_SIZE * block_pos);

    while(len > already_count) {
        memcpy(buffer_now, block_ptr, BLOCK_SIZE);
        while(offset < BLOCK_SIZE) {
            *(buffer_now + offset) = *text_ptr;
            text_ptr ++;
            already_count ++;
            if (len == already_count) {
                break;
            }
            offset ++;
        }

        memcpy(block_ptr, buffer_now, BLOCK_SIZE);

        if (offset == BLOCK_SIZE && len != already_count) {
            offset = 0;
            block_pos = fat_ptr -> id;
            if (block_pos == END) {
                fat* free_fat = (fat*) (my_v_hard + BLOCK_SIZE);
                for(int i = 0; i < (int)(SIZE /BLOCK_SIZE); i ++) {
                    if (free_fat[i].id == FREE) {
                        block_pos = i;
                    }
                }
                if (block_pos == END) {
                    printf("error in do_write.\n" );
                    return -1;
                }
                block_ptr = (unsigned char*)(my_v_hard + BLOCK_SIZE * block_pos);
                fat_ptr->id = block_pos;
                fat_ptr = (fat*)(my_v_hard + BLOCK_SIZE) + block_pos;
                fat_ptr->id = END;
            }
            else {
                    block_ptr = (unsigned char*)(my_v_hard + BLOCK_SIZE * block_pos);
                    fat_ptr = (fat*)(my_v_hard + BLOCK_SIZE) + block_pos;
                }
            }
    }

    open_file_list[fd].count += len;
    if (open_file_list[fd].open_file_fcb.length < open_file_list[fd].count) {
        open_file_list[fd].open_file_fcb.length = open_file_list[fd].count;
    }

    if (wstyle == 1 || (wstyle == 2 && open_file_list[fd].open_file_fcb.attribute == 0)) {
        int temp;
        offset = (int)open_file_list[fd].open_file_fcb.length;
        fat_ptr = (fat*) (my_v_hard + BLOCK_SIZE) + open_file_list[fd].open_file_fcb.first;

        while(offset >= BLOCK_SIZE) {
            block_pos = fat_ptr->id;
            offset -= BLOCK_SIZE;
            fat_ptr = (fat*)(my_v_hard + BLOCK_SIZE) + block_pos;
        }   

        while(true) {
            if (fat_ptr->id != END) {
                temp = fat_ptr->id;
                fat_ptr->id = FREE;
                fat_ptr = (fat*) (my_v_hard + BLOCK_SIZE) + temp;
            }
            else {
                fat_ptr = FREE;
                break;
            }
        }
        fat_ptr = (fat*) (my_v_hard + BLOCK_SIZE) + block_pos;
        fat_ptr->id = END;
    }

    fat* second_fat = (fat*)(my_v_hard + BLOCK_SIZE * 3);
    fat* first_fat = (fat*)(my_v_hard + BLOCK_SIZE);
    memcpy(second_fat, first_fat, BLOCK_SIZE * 2);
    return len;
}

    if (fd >= MAX_OPEN_FILE || fd < 0) {

        else break;
    }
    char inp[SIZE];
    printf("please input you text, and -1 for finish!\n");

int my_open(char *filename) {
    static char buffer[SIZE];
    ptr_current_dir = 0;
    do_read(ptr_current_dir - open_file_list, (int)ptr_current_dir->open_file_fcb.length, buffer);

    fcb *fcb_ptr = (fcb*)buffer;
    int flag = 0;
    int i;
    for(i = 0; i < (int)(ptr_current_dir->open_file_fcb.length / sizeof(fcb)); i ++) {
        if (strcmp(fcb_ptr->filename, filename) == 0 && fcb_ptr->attribute == 1) {
            flag = 1;
            break;
        }
        fcb_ptr++;
    }
    if (flag == 0) {
        printf("file is not exist!!\n");
        return -1;
    }
    int free_fd = - 1;
    for (int j = 0; j < MAX_OPEN_FILE; j ++) {
        if (open_file_list[j].t_openfile == 0) {
            open_file_list[j].t_openfile = 1;
            free_fd = j;
            break;
        }
    }
    if (free_fd == - 1) {
        printf("error in my_open:MAX_OPEN_FILE!!\n");
        return -1;
    }

    open_file_list[free_fd].open_file_fcb.attribute = 1;
    open_file_list[free_fd].count = 0;
    open_file_list[free_fd].open_file_fcb.date = fcb_ptr->date;
    open_file_list[free_fd].open_file_fcb.time = fcb_ptr->time;
    open_file_list[free_fd].open_file_fcb.length = fcb_ptr->length;
    open_file_list[free_fd].open_file_fcb.first = fcb_ptr->first;
    open_file_list[free_fd].t_openfile = 1;
    strcpy_s(open_file_list[free_fd].open_file_fcb.filename, strlen(fcb_ptr->filename), fcb_ptr->filename);
    strcat(strcpy((char*)open_file_list[free_fd].dir, (char*)ptr_current_dir->dir), filename);
    strcpy_s(open_file_list[free_fd].open_file_fcb.extname, strlen(fcb_ptr->extname), fcb_ptr->extname);
    open_file_list[free_fd].t_openfile = 1;
    open_file_list[free_fd].dir_no = ptr_current_dir->open_file_fcb.first;
    open_file_list[free_fd].dir_off = i;
    open_file_list[free_fd].fcb_state = 0;

    ptr_current_dir = open_file_list + free_fd;

    return free_fd;
}
